Стилизация radio

html

          <label>
              <input class="radio" type="radio">
              <span class="radio-style"></span> radio
          </label>
                  
          <label>
              <input class="radio" type="radio" checked>
              <span class="radio-style"></span> checked radio
          </label>
                  
          <label>
              <input class="radio" type="radio" disabled>
              <span class="radio-style"></span> disabled radio
          </label>
                  
          <label>
              <input class="radio" type="radio" checked disabled>
              <span class="radio-style"></span> checked disabled radio
          </label>

css

          label {
            display: flex;
            align-items: center;
            margin-bottom: 20px;
            /* Отступ для check-style*/
            padding-left: 30px;
          }
           
          .radio {
            position: absolute;
            width: 1px;
            height: 1px;
            overflow: hidden;
            clip: rect(0 0 0 0);
          }
           
          .radio-style {
            /* Любые стили по макету */
            position: absolute;
            width: 16px;
            height: 16px;
            border: 2px solid #000;
            border-radius: 50%;
            /* Отступ */
            margin-left: -30px;
          }
           
          /* Стили для checked radio */
          .radio:checked + .radio-style::before {
            content: "";
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
            width: 8px;
            height: 8px;
            border-radius: 50%;
            background-color: #000;
          }
           
          /* Стилизация иконкой, вместо before
          .radio:checked + .radio-style {
            background-image: url();
          } */
           
          /* Стили для disabled radio */
          .radio:disabled + .radio-style {
            border-color: #ccc;
          }
           
          /* Стили для checked disabled radio */
          .radio:checked:disabled + .radio-style::before {
            background-color: #ccc;
          }
           
          /* Стили, когда radio в фокусе */
          .radio:focus + .radio-style {
            box-shadow: 0 0 0 1px #000;
          }
