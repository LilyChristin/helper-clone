Стилизация checkbox

html

          <label>
            <input class="check-box" type="checkbox">
            <span class="check-style"></span> checkbox
          </label>
           
          <label>
            <input class="check-box" type="checkbox" checked>
            <span class="check-style"></span> checked checkbox
          </label>
           
          <label>
            <input class="check-box" type="checkbox" disabled>
            <span class="check-style"></span> disabled checkbox
          </label>
           
          <label>
            <input class="check-box" type="checkbox" checked disabled>
            <span class="check-style"></span> checked disabled checkbox
          </label>

css

          label {
            display: block;
            margin-bottom: 20px;
            /* Отступ для check-style*/
            padding-left: 20px;
          }
           
          .check-box {
            position: absolute;
            width: 1px;
            height: 1px;
            overflow: hidden;
            clip: rect(0 0 0 0);
          }
           
          .check-style {
            /* Любые стили по макету */
            position: absolute;
            width: 16px;
            height: 16px;
            border: 2px solid #000;
            /* Отступ */
            margin-left: -20px;
          }
           
          /* Стили для checked checkbox */
          .check-box:checked + .check-style::before {
            content: "";
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
            width: 8px;
            height: 8px;
            background-color: #000;
          }
           
          /* Стилизация иконкой, вместо before
          .check-box:checked + .check-style {
            background-image: url();
          } */
           
          /* Стили для disabled checkbox */
          .check-box:disabled + .check-style {
            border-color: #ccc;
          }
           
          /* Стили для checked disabled checkbox */
          .check-box:checked:disabled + .check-style::before {
            background-color: #ccc;
          }
           
          /* Стили, когда checkbox в фокусе */
          .check-box:focus + .check-style {
            box-shadow: 0 0 0 1px #000;
          }
